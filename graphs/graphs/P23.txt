#include<bits/stdc++.h>
using namespace std;

class Solution{
public:
    vector<int> KahnsAlgo(vector<vector<int>> &graph, int v){
        vector<int> inDegree(v,0);
        
        for(int i=0;i<v;i++){
            for(int it:graph[i]){
                inDegree[it]++;
            }
        }
        queue<int> q;
        for(int i=0;i<v;i++){
            if(inDegree[i]==0){
                q.push(i);
            }
        }
        
        vector<int> topo;
        while(!q.empty()){
            v = q.front(); q.pop();
            topo.push_back(v);
            for(int it:graph[v]){
                inDegree[it]--;
                if(inDegree[it]==0){
                    q.push(it);
                }
            }
        }
        return topo;
    }
};

vector<vector<int>> getGraph(string s,int v){
    vector<vector<int>> graph(v);
    vector<int> temp;
    string num  = "";
    for(char c:s){
        if(isdigit(c) || c=='-'){
            num+=c;
        }else if(!num.empty()){
            temp.push_back(stoi(num));
            num = "";
            if(temp.size()==2){
                int v1 = temp[0],v2 = temp[1];
                graph[v1].push_back(v2);
                temp.clear();
            }
        }
    }
    return graph;
}

void printVec(vector<int> &v){
    cout<<"[";
    for(int i=0;i<v.size();i++){
        cout<<v[i]<<" ";
    }
    cout<<"]";
}

int main(){
    int v;
    cin>>v;
    cin.ignore(numeric_limits<streamsize>::max(), '\n');
    string s;
    getline(cin,s);
    vector<vector<int>> graph = getGraph(s,v);
    Solution S;
    vector<int> order = S.KahnsAlgo(graph,v);
    printVec(order);
    return 0;
}
