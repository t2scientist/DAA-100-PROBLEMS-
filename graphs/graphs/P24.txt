#include<bits/stdc++.h>
using namespace std;

vector<vector<int>> makeDependenciesGraph(int numCourses,vector<vector<int>>& prerequisites){
    vector<vector<int>> dependenciesGraph(numCourses);
    
    for(int i=0;i<prerequisites.size();i++){
        dependenciesGraph[prerequisites[i][1]].push_back(prerequisites[i][0]);
    }
    return dependenciesGraph;
}


class Solution {
public:
    void dfs(int src,vector<vector<int>> &graph,vector<int> &visited,vector<int>& recursionStack,bool *cycleExists){
        
        if(*cycleExists) return;
        
        visited[src] = 1;
        recursionStack[src] = 1;
        for(auto n:graph[src]){
            if(!visited[n]){
                dfs(n,graph,visited,recursionStack,cycleExists);
            }
            else if(recursionStack[n]==1){
                *cycleExists = true;
            }
        }
        recursionStack[src] = 0;
    }
    
    bool isCycleinDAG(vector<vector<int>> &graph,int v){
        vector<int> visited(v,0),recursionStack(v,0);
        bool cycleExists = false;
        for(int i=0;i<v;i++){
            if(!visited[i]){
                dfs(i,graph,visited,recursionStack,&cycleExists);
            }
            if(cycleExists){
                return true;
            }
        }
        return false;
    }
    
    bool canFinish(int numCourses, vector<vector<int>>& prerequisites) {
        vector<vector<int>> dependenciesGraph = makeDependenciesGraph(numCourses,prerequisites);
        return !isCycleinDAG(dependenciesGraph,numCourses);
    }
};

vector<vector<int>> parseToVector(string s) {
    vector<vector<int>> result;
    vector<int> temp;
    string num = "";

    for (char c : s) {
        if (isdigit(c) || c == '-') {
            num += c; 
        } else if (!num.empty()) {
            temp.push_back(stoi(num));
            num = "";
        }

        if (c == ']') {
            if (!temp.empty()) {
                result.push_back(temp);
                temp.clear();
            }
        }
    }

    return result;
}

int main(){
    int numCourses;
    cin>>numCourses;
    cin.ignore(numeric_limits<streamsize>::max(), '\n');
    string s;
    getline(cin,s);
    vector<vector<int>> prerequisites = parseToVector(s);
    Solution S;
    cout<<(S.canFinish(numCourses,prerequisites)?"true":"false");
    return 0;
}
