// dict = {"baa","abcd","abca","cab","cad"}
#include<bits/stdc++.h>
using namespace std;

void printVec(vector<string> &v){
    cout<<"[";
    for(int i=0;i<v.size();i++){
        cout<<v[i]<<" ";
    }
    cout<<"]";
}

vector<string> getWordsList(string s){
    vector<string> words;
    string word ="";
    for(char c:s){
        if(isalpha(c)){
            word+=c;
        }else if(!word.empty()){
            words.push_back(word);
            word = "";
        }
    }
    if(!word.empty()){
            words.push_back(word);
    }
    
    return words;
}

void printAdjList(vector<vector<int>> &adjList) {
    cout << "[\n";
    for (auto &v : adjList) {
        cout << "[";
        for (auto x : v) cout << x << " ";
        cout << "]\n";
    }
    cout << "]\n";
}


class Solution{
public:
    string topoSort(vector<vector<int>> &graph,int v){
        queue<int> q;
        vector<int> inDegree(v,0);
        string order="";
        for(int i=0;i<v;i++){
            for(auto n:graph[i]){
                inDegree[n]++;
            }
        }
        
        for(int i=0;i<v;i++){
            if(inDegree[i]==0){
                q.push(i);
            }
        }
        
        while(!q.empty()){
            int V = q.front(); q.pop();
            order+=char(V+'a');
            for(auto n:graph[V]){
                inDegree[n]--;
                if(inDegree[n]==0){
                    q.push(n);
                }
            }
        }
        return order;
    }
    string findOrder(vector<string> &words,int k){
        string order = "";
        unordered_map<char,int> um;
        for(int i=0;i<k;i++){
            um['a'+i] = i;
        }
        
        printVec(words);
        
        vector<vector<int>> graph(k);
        int l = words.size();
        string s1,s2;
        for(int i=0;i<l-1;i++){
            s1 = words[i]; s2 = words[i+1];
            int maxIter = min(s1.length(),s2.length());
            for(int j=0;j<maxIter;j++){
                if(s1[j]!=s2[j]){
                    graph[um[s1[j]]].push_back(um[s2[j]]);
                    break;
                }
            }
        }
        
        printAdjList(graph);
        
        return topoSort(graph,k);
    }    
};


int main(){
    Solution S;
    int k;
    cin>>k;
    cin.ignore();
    string s;
    getline(cin,s);
    vector<string> words = getWordsList(s);
    string order = S.findOrder(words,k);
    cout<<order;
    return 0;
}
