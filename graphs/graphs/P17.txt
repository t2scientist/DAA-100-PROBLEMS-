#include<bits/stdc++.h>
using namespace std;

class Solution {
public:
    int numEnclaves(vector<vector<int>>& grid) {
        int rows = grid.size(),cols = grid[0].size();
        queue<pair<int,int>> q;
        
        for(int i=0;i<rows;i++){
            for(int j=0;j<cols;j++){
                if(i==0 || i==rows-1){
                    if(grid[i][j]==1){
                        q.push({i,j});
                        grid[i][j] = 2;
                    }
                }else{
                    if(j==0 || j==cols-1){
                        if(grid[i][j]==1){
                            q.push({i,j});
                            grid[i][j] = 2;
                        }
                    }
                }
            }
        }
        
        int dx[] = {1,0,-1,0},dy[] = {0,-1,0,1};
        while(!q.empty()){
            auto [x,y] = q.front(); q.pop();
            for(int k=0;k<4;k++){
                int i = x+dx[k],j = y+dy[k];
                if(i>=0 && i<rows && j>=0 && j<cols){
                    if(grid[i][j]==1){
                        q.push({i,j});
                        grid[i][j] = 2;
                    }
                }
            }
        }
        int numofEnclaves = 0;
        for(int i=0;i<rows;i++){
            for(int j=0;j<cols;j++){
                if(grid[i][j]==1){
                    numofEnclaves++;
                }
            }
        }
        return numofEnclaves;
    }
};

vector<vector<int>> getBoard(){
    vector<vector<int>> board;
    string s;
    getline(cin,s);
    vector<int> temp;
    for(char c:s){
        if(c==']'  && !temp.empty()){
            board.push_back(temp);
            temp.clear();
        }
        if(c=='1' || c=='0'){
            temp.push_back((int)(c-'0'));
        }
    }
    return board;
}

    int rows = board.size(),cols = board[0].size();
    for(int i=0;i<rows;i++){
        cout<<"[";
        for(int j=0;j<cols;j++){

    }
