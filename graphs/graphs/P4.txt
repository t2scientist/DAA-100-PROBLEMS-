// Input: grid - [ [1,0,1] , [1,1,0] , [1,0,0] ]
#include<bits/stdc++.h>
using namespace std;

class Solution {
public:
    vector<vector<int>> updateMatrixSingleSourceBFS(vector<vector<int>>& mat) {
        int rows = mat.size(),cols = mat[0].size();
        vector<vector<int>> dist_mat(rows,vector<int>(cols));
        int dx[] ={1,0,-1,0},dy[]={0,-1,0,1};
        for(int i=0;i<rows;i++){
            for(int j=0;j<cols;j++){
                if(mat[i][j]==1){
                    dist_mat[i][j] = 0;
                }else{
                    bool cond = true;
                    queue<tuple<int,int,int>> q;
                    for(int k=0;k<4;k++){
                        int di = i+dx[k],dj = j+dy[k];
                        if(di>=0 && di<rows && dj>=0 && dj<cols && mat[di][dj]==1){
                            dist_mat[i][j] = 1;
                            cond = false;
                            break;
                        }else if(di>=0 && di<rows && dj>=0 && dj<cols && mat[di][dj]==0){
                            q.push(tuple{di,dj,1});
                        }
                    }
                    while(!q.empty() && cond){
                        auto [x,y,d] = q.front();
                        q.pop();
                        for(int k=0;k<4;k++){
                            int di = x+dx[k],dj = y+dy[k];
                            if(di>=0 && di<rows && dj>=0 && dj<cols && mat[di][dj]==1){
                                dist_mat[i][j] = d+1;
                                cond = false;
                                break;
                            }else if(di>=0 && di<rows && dj>=0 && dj<cols && mat[di][dj]==0){
                                q.push(tuple{di,dj,d+1});
                            }
                        }
                    }
                    
                }
            }
        }
        return dist_mat;
        
    }
    
    
    
    vector<vector<int>> updateMatrixMultiSourceBFS(vector<vector<int>> &mat){
        int rows = mat.size(),cols = mat[0].size();
        queue<tuple<int,int,int>> q;
        vector<vector<int>> visited(rows,vector<int>(cols)),dist_matrix(rows,vector<int>(cols));
        for(int i=0;i<rows;i++){
            for(int j=0;j<cols;j++){
                if(mat[i][j]==1){
                    q.push(tuple{i,j,0});
                    visited[i][j] = 1;
                    dist_matrix[i][j] = 0;
                }
            }
        }
        int dx[] ={1,0,-1,0},dy[]={0,-1,0,1};
        while(!q.empty()){
            auto [i,j,d] = q.front();
            q.pop();
            for(int k=0;k<4;k++){
                int di = i+dx[k],dj = j+dy[k];
                if(di>=0 && di<rows && dj>=0 && dj<cols && mat[di][dj]==0 && !visited[di][dj]){
                    dist_matrix[di][dj] = d+1;
                    visited[di][dj] = 1;
                    q.push(tuple{di,dj,d+1});
                }
            }
        }
        return dist_matrix;
    }

};


vector<vector<int>> getGrid(){
    string s,num="";
    getline(cin,s);
    int rows = 0,cols = 0;
    for(char c:s){
        if(c=='['){
            rows++;
        }
    }
    rows--;
    vector<int> v;
    for(char c:s){
        if(isdigit(c)){
            num+=c;
        }else if(!num.empty()){
            v.push_back(stoi(num));
            num = "";
        }
    }
    
    if(!num.empty()){
        v.push_back(stoi(num));
    }
    cols = v.size()/rows;
    int index = 0;
    vector<vector<int>> grid(rows,vector<int>(cols));
    for(int i=0;i<rows;i++){
        for(int j=0;j<cols;j++){
            grid[i][j] = v[index++];
        }
    }
    return grid;
};

void printMat(vector<vector<int>> &mat){
    int rows = mat.size(),cols = mat[0].size();
    cout<<"[\n";
    for(int i=0;i<rows;i++){
        cout<<"[";
        for(int j=0;j<cols;j++){
            cout<<mat[i][j]<<" ";
        }
        cout<<"]\n";
    }
    cout<<"]";
}

int main(){
    Solution S;
    auto grid = getGrid();
    vector<vector<int>> zero_one_Matrix = S.updateMatrixMultiSourceBFS(grid);
    printMat(zero_one_Matrix);
    return 0;
}
