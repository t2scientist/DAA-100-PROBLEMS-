#include<bits/stdc++.h>
using namespace std;

void mergeInplace(vector<int> &nums1, vector<int> &nums2){
    int m = nums1.size() - nums2.size(), n = nums2.size(), i=m-1,j=n-1,k=m+n-1;
    
    while(i>=0 && j>=0){
        if(nums1[i]>nums2[j]){
            nums1[k--] = nums1[i--];
        }else{
            nums1[k--] = nums2[j--];
        }
    }
    while(i>=0){
        nums1[k--] = nums1[i--];
    }
    
    while(j>=0){
        nums1[k--] = nums2[j--];
    }
}

void printarr(vector<int> &a){
    cout<<"[";
    for(int i=0;i<a.size();i++){
        cout<<a[i]<<", ";
    }
    cout<<"]";
}

int main(){
    vector<int> nums1 = {-5,-2,4,5,0,0,0};
    vector<int> nums2 = {-3,1,8};
    
    mergeInplace(nums1,nums2);
    printarr(nums1);
    return 0;
}
