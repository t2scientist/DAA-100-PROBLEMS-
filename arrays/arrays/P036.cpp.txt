#include<bits/stdc++.h>
using namespace std;

vector<int> findMissingAndRepeatedValues(vector<int>& nums) {
    int n = nums.size(),sum=0,sq_sum=0;
    for(int i=0;i<n;i++){
        sum+=nums[i];
        sq_sum+=nums[i]*nums[i];
    }
    int sum_n = (n*(n+1))/2;
    int sq_sum_n = (n*(n+1)*(2*n+1))/6;
    int a_b_sum = (sq_sum-sq_sum_n)/(sum-sum_n);
    int a_b_diff = sum-sum_n;
    int a = (a_b_sum+a_b_diff)/2;
    int b =  a_b_sum-a;
    return {a,b};
}

void printarr(vector<int> &a){
    cout<<"[";
    for(int i=0;i<a.size();i++){
        cout<<a[i]<<", ";
    }
    cout<<"]";
}

vector<int> getarr(){
    string s,num="";
    getline(cin,s);
    vector<int> arr;
    for(char c:s){
        if(isdigit(c) || c=='-'){
            num+=c;
        }else if(!num.empty()){
            arr.push_back(stoi(num));
            num="";
        }
    }
    if(!num.empty()){
        arr.push_back(stoi(num));
    }
    return arr;
}

int main(){
    vector<int> nums = getarr();
    vector<int> x = findMissingAndRepeatedValues(nums);
    cout<<x[0]<<" "<<x[1];
    return 0;
}
